trigger = {
	fromfrom = {
		OR = {
			is_country_type = gray
			is_same_value = ROOT
		}
	}
	RG_gray_army_active_trigger = yes
	NOR = {
		RG_gray_not_avaliable_trigger = yes
	}
	is_same_value = event_target:gray_owner
	any_owned_army = {
		army_type = gray_army
		OR = {
			exists = planet
			AND = {
				exists = fleet
				fleet = {
					num_ships > 0
				}
			}
		}
	}
}

	set_country_flag = gray_diplomacy_engaged
}
after = {
	hidden_effect = {
		remove_country_flag = gray_diplomacy_engaged
		reset_gray_cloth = yes
	}
}

	# 开启隐形
	name = rg_event.4.e
	response_text = rg_cloak.a.response
	trigger = {
		any_owned_fleet = {
			AND = {
				is_ship_size = RG_armed_transport_ship
				is_cloaked = no
			}
		}
	}
	hidden_effect = {
		every_owned_fleet = {
			limit = {
				any_owned_ship = {
					is_ship_size = RG_armed_transport_ship
				}
			}
			set_cloaking_active = yes
		}
	}
}
